SELECT CONCAT('+' || GNAME,SYS_CONNECT_BY_PATH(ANAME,'/')) FULL_PATH, SYSTEM_CREATED, ALIAS_DIRECTORY, FILE_TYPE FROM
(SELECT B.NAME GNAME,A.PARENT_INDEX PINDEX, A.NAME ANAME, A.REFERENCED_INDEX RINDEX,
A.SYSTEM_CREATED, A.ALIAS_DIRECTORY, C.TYPE FILE_TYPE
FROM V$ASM_ALIAS A, V$ASM_DISKGROUP B, V$ASM_FILE C
WHERE A.GROUP_NUMBER=B.GROUP_NUMBER
AND A.GROUP_NUMBER=C.GROUP_NUMBER(+)
AND A.FILE_NUMBER=C.FILE_NUMBER(+)
AND A.FILE_INCARNATION =C.INCARNATION(+)
)
START WITH (MOD(PINDEX, POWER(2,24))) =0
CONNECT BY PRIOR RINDEX=PINDEX;
