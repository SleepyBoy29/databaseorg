Set colsep '|'
SET VERIFY OFF
SET LINE 40
COLUMN KILL_JOB FORMAT A40
COLUMN USERNAME FORMAT A40

SELECT 'EXEC DBMS_SCHEDULER.STOP_JOB ('|| R.JOB ||',TRUE);' AS KILL_JOB,S.SID,S.USERNAME
FROM DBA_JOBS J, DBA_JOBS_RUNNING R, V$SESSION S
WHERE J.JOB=R.JOB
AND R.SID=S.SID
AND S.SID=R.SID
AND S.USERNAME='&USERNAME';
