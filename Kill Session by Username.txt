set colsep '|'
SET LINESIZE 150
SET VERIFY OFF
COLUMN COMMAND FORMAT A45
COLUMN USERNAME FORMAT A15
COLUMN OSUER FORMAT A25
COLUMN MACHINE FORMAT A25
COLUMN PROGRAM FORMAT A15
SELECT 'ALTER SYSTEM KILL SESSION '''|| SID ||','||SERIAL#||''';' AS COMMAND,USERNAME,OSUSER,PROGRAM,MACHINE
FROM V$SESSION WHERE USERNAME='&USERNAME';
