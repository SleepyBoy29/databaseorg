Check Active Sessions in Oracle 9i Database.

SET COLSEP '|'
COLUMN SERIAL# FORMAT 99999
COLUMN USERNAME FORMAT A17
COLUMN MACHINE FORMAT A22
COLUMN OSUSER FORMAT A15
COLUMN SID FORMAT 9999
COLUMN SPID FORMAT 9999
COLUMN STATUS FORMAT A10
COLUMN PROGRAM FORMAT A15
COLUMN BLOCKING_SESSION FORMAT 9999
SET LINESIZE 140
SET VERIFY OFF

SELECT S.SID,S.SERIAL#,P.SPID "SPID",S.USERNAME,S.STATUS,S.OSUSER,S.MACHINE,S.PROGRAM
FROM V$SESSION S,V$PROCESS P
WHERE S.PADDR=P.ADDR
AND S.USERNAME IS NOT NULL
ORDER BY S.USERNAME;
