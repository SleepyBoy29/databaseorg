set colsep '|'

SET LINESIZE 150

SET VERIFY OFF

COLUMN USERNAME FORMAT A20

COLUMN MACHINE FORMAT A20

COLUMN COMMAND FORMAT A65

COLUMN PROGRAM FORMAT A20

SELECT 'EXEC DBMS_SYSTEM.SET_SQL_TRACE_IN_SESSION('|| SID ||','||SERIAL#||',TRUE);' AS COMMAND,USERNAME,PROGRAM,MACHINE FROM V$SESSION WHERE SID=&SID;

SELECT 'EXEC DBMS_MONITOR.SESSION_TRACE_ENABLE(SESSION_ID=>'|| SID ||',SERIAL_NUM=>'||SERIAL#||');' AS COMMAND,USERNAME,PROGRAM,MACHINE FROM V$SESSION WHERE SID=&SID;

SELECT 'oradebug setorapid &sid;' from dual;

select 'oradebug event 10046 trace name context forever, level 12;' AS COMMAND,USERNAME,PROGRAM,MACHINE FROM V$SESSION WHERE SID=&SID;
